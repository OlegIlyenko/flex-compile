71      protected void sortComponents(List<Component> components, List<Component> resultComponents, Map<Component, Set<Component>> componentDependencies, Stack<Component> componentStack) {
72          for (Component c : components) {
73              if (resultComponents.contains(c)) {
74                  continue; // component already processed
75              }
76  
77              if (componentStack.contains(c)) {
78                  throw new IllegalStateException("Project contains recursive dependencies. Component: " + c.getName());
79              }
80  
81              if (c.getDependencies() == null || c.getDependencies().size() == 0) {
82                  resultComponents.add(c);
83                  componentDependencies.put(c, new HashSet<Component>());
84              } else {
85                  List<Component> cd = new ArrayList<Component>();
86                  List<Component> comps = new ArrayList<Component>();
87                  for (Dependency d : c.getDependencies()) {
88                      comps.add(d.getComponent());
89                  }
90  
91                  componentStack.push(c);
92                  sortComponents(comps, resultComponents, componentDependencies, componentStack);
93                  componentStack.pop();
94  
95                  resultComponents.add(c);
96                  componentDependencies.put(c, collectDependencies(comps, componentDependencies));
97              }
98          }
99      }

101     protected Set<Component> collectDependencies(List<Component> dependencies, Map<Component, Set<Component>> componentDependencies) {
102         Set<Component> result = new HashSet<Component>();
103 
104         for (Component c : dependencies) {
105             if (componentDependencies.containsKey(c)) {
106                 Set<Component> d = collectDependencies(new ArrayList<Component>(componentDependencies.get(c)), componentDependencies);
107                 result.addAll(d);
108             }
109 
110             result.add(c);
111         }
112 
113         return result;
114     }

124     public synchronized List<CompilationResults> compile(String targetName, File projectFile, String dstDir) {
125         Project project;
126         List<Component> sortedComponentList = null;
127         Map<Component, Set<Component>> componentDependencies = null;
128 
129         ProjectCacheEntry cacheEntry = PROJECT_CACHE.get(projectFile);
130 
131         boolean projectChanges = false;
132         try {
133             if (cacheEntry == null || projectDao.isProjectModified(cacheEntry.getProject(), projectFile.toURI().toURL())) {
134                 try {
135                     project = projectDao.getProject(projectFile.toURI().toURL());
136                 } catch (Exception e) {
137                     throw new RuntimeException("Error in project definition.", e);
138                 }
139 
140                 sortedComponentList = new ArrayList<Component>();
141                 componentDependencies = new HashMap<Component, Set<Component>>();
142                 sortComponents(project.getComponents(), sortedComponentList, componentDependencies, new Stack<Component>());
143 
144                 PROJECT_CACHE.put(projectFile, new ProjectCacheEntry(project, sortedComponentList, componentDependencies));
145                 applicationCache.clear();
146                 componentCache.clear();
147                 projectChanges = true;
148             } else {
149                 project = cacheEntry.getProject();
150                 sortedComponentList = cacheEntry.getSortedComponents();
151                 componentDependencies = cacheEntry.getComponentDependencies();
152             }
153         } catch (MalformedURLException e) {
154             e.printStackTrace();
155             throw new RuntimeException(e);
156         }
157 
158         List<CompilationResults> compilationResultses = new ArrayList<CompilationResults>();
159         long startTime = System.currentTimeMillis();
160 
161         fireEvent(new CompilerEvent(this, "Starting project '" + project.getName() + "'...", CompilerEvent.EventType.ProjectStart, project.toString()));
162 
163         boolean hasErrors = false;
164         StringBuilder projectProtocol = new StringBuilder();
165         for (Component comp : sortedComponentList) {
166             CompilationResults compilationResults = compile(targetName, project, comp, projectFile.getParentFile(), dstDir, componentDependencies, projectChanges);
167             compilationResultses.add(compilationResults);
168 
169             projectProtocol.append("------------------------------------------------------------- ").append(comp.toString()).append("\n");
170             projectProtocol.append(compilationResults.getCompileProtocol()).append("\n\n");
171 
172             if (compilationResults.isHasErrors()) {
173                 hasErrors = true;
174                 break;
175             }
176         }
177 
178         long processingTime = System.currentTimeMillis() - startTime;
179 
180         CompilerEvent event;
181         if (hasErrors) {
182             event = new CompilerEvent(this, "Flex project compilation fail.", CompilerEvent.EventType.ProjectFinishErrors, project.toString());
183         } else {
184             event = new CompilerEvent(this, "Flex project compilation success.", CompilerEvent.EventType.ProjectFinishSuccess, project.toString());
185         }
186         event.setCompilationProtocol(projectProtocol.toString());
187         event.setCompilationTime(processingTime);
188         fireEvent(event);
189 
190         return compilationResultses;
191     }

193     protected File getArtifactOutputFile(Artefact artefact, File dstDir) {
194         File outputFile = null;
195 
196         String targetFileName = artefact.getFileName().substring(0, artefact.getFileName().lastIndexOf(".")) + ".swf";
197         switch (artefact.getType()) {
198             case application:
199                 outputFile = new File(dstDir, File.separator + targetFileName);
200                 break;
201             case module:
202                 File swfTargetDir = new File(dstDir, Constants.ComponentFolders.MODULE_TARGET_DIR);
203                 if (!swfTargetDir.exists()) {
204                     swfTargetDir.mkdirs();
205                 }
206                 outputFile = new File(swfTargetDir, targetFileName);
207                 break;
208             case css:
209                 File swfTargetCssDir = new File(dstDir, Constants.ComponentFolders.CSS_TARGET_DIR);
210                 if (!swfTargetCssDir.exists()) {
211                     swfTargetCssDir.mkdirs();
212                 }
213                 outputFile = new File(swfTargetCssDir, targetFileName);
214                 break;
215         }
216 
217         return outputFile;
218     }

220     protected CompilationResults compile(String targetName, Project project, Component component, File projectDir, String dstDir, Map<Component, Set<Component>> componentDependencies, boolean projectChanges) {
221         StringBuilder protocol = new StringBuilder();
222 
223         long startTime = System.currentTimeMillis();
224 
225         String currentPath = targetName;
226 
227         // detect changes
228         if (targetName != null && !projectChanges) {
229             File swf = null;
230             if (component instanceof org.oleg.fcs.project.model.Application) {
231                 org.oleg.fcs.project.model.Application app = (org.oleg.fcs.project.model.Application) component;
232 
233                 Artefact appArt = app.getArtefacts().get(0);
234                 for (Artefact a : app.getArtefacts()) {
235                     if (a.getType() == Artefact.Type.application && a.getFileName().equals(targetName + ".mxml")) {
236                         appArt = a;
237                         break;
238                     }
239                 }
240 
241                 String appArtFileName = appArt.getFileName().substring(0, appArt.getFileName().lastIndexOf(".")) + ".swf";
242 
243                 if (appArt.getType() == Artefact.Type.application) {
244                     swf = new File(dstDir, appArtFileName);
245                 } else if (appArt.getType() == Artefact.Type.module) {
246                     swf = new File(dstDir, appArtFileName);
247                 } else if (appArt.getType() == Artefact.Type.css) {
248                     swf = new File(dstDir, Constants.ComponentFolders.CSS_TARGET_DIR + File.separator + appArtFileName);
249                 }
250             } else {
251                 swf = new File(dstDir, Constants.ComponentFolders.SWC_TARGET_DIR + File.separator + component.getName() + ".swc");
252             }
253 
254             if (swf.exists()) {
255                 Date resDate = new Date(swf.lastModified());
256                 if (!isModificationsPresent(new File(projectDir, component.getName()), resDate)) {
257                     protocol.append("No changes... Finishing.");
258 
259                     CompilerEvent event = new CompilerEvent(this, "No Chnages", CompilerEvent.EventType.Success, component.toString());
260                     event.setCompilationProtocol(protocol.toString());
261                     fireEvent(event);
262 
263                     return new CompilationResults(protocol.toString(), false, false, project.getName(), component.getName());
264                 }
265             }
266         }
267 
268         Report report = null;
269         String currentBuildTarget = null;
270         List<File> outputFiles = new ArrayList<File>();
271 
272         if (!(component instanceof org.oleg.fcs.project.model.Application)) {
273             protocol.append("Starting library '" + component.getName() + "'...").append("\n");
274             fireEvent(new CompilerEvent(this, "Starting...", CompilerEvent.EventType.Started, component.toString()));
275 
276             File swcTargetDir = new File(dstDir, Constants.ComponentFolders.SWC_TARGET_DIR);
277             if (!swcTargetDir.exists()) {
278                 swcTargetDir.mkdirs();
279             }
280 
281             String flexComponentCacheKey = project.toString() + component.toString();
282             Library flexComponent = componentCache.get(flexComponentCacheKey);
283             if (flexComponent == null) {
284                 flexComponent = new Library();
285 
286                 if (conf.getBoolean(ConfigProperties.SHOW_PROGRESS, false)) {
287                     flexComponent.setProgressMeter(progressTracker);
288                 }
289 
290                 flex2.tools.oem.Configuration configuration = flexComponent.getDefaultConfiguration();
291                 configure(configuration, new File(projectDir, component.getName()), new File(dstDir), component, componentDependencies.get(component), project, projectDir);
292 //                configuration.setIncludes(new String[] {"*.*"});
293                 flexComponent.setConfiguration(configuration);
294 
295                 File mxmlComp = new File(projectDir, component.getName() + File.separator + Constants.ComponentFolders.MXML_DIR);
296                 if (mxmlComp.exists()) {
297                     flexComponent.addComponent(mxmlComp);
298                 }
299 
300                 File srcComp = new File(projectDir, component.getName() + File.separator + Constants.ComponentFolders.SRC_DIR);
301                 if (srcComp.exists()) {
302                     flexComponent.addComponent(srcComp);
303                     addLibraryAssets(srcComp, "", flexComponent);
304                 }
305 
306                 flexComponent.setOutput(new File(swcTargetDir, component.getName() + ".swc"));
307                 componentCache.put(flexComponentCacheKey, flexComponent);
308             }
309 
310             outputFiles.add(new File(swcTargetDir, component.getName() + ".swc"));
311 
312             protocol.append("Compiling...").append("\n");
313             fireEvent(new CompilerEvent(this, "Compiling...", CompilerEvent.EventType.Compilation, component.toString()));
314 
315             try {
316                 flexComponent.build(true);
317             } catch (IOException e) {
318                 log.error("Error during copilation", e);
319                 protocol.append("Technical error: ").append(e.getMessage());
320                 return new CompilationResults(protocol.toString(), true, true, project.getName(), component.getName());
321             }
322 
323             report = flexComponent.getReport();
324             currentBuildTarget = component.toString();
325         } else { // Application
326             org.oleg.fcs.project.model.Application app = (org.oleg.fcs.project.model.Application) component;
327             for (Artefact artefact : app.getArtefacts()) {
328                 if (targetName != null && artefact.getType() == Artefact.Type.application && !artefact.getFileName().equals(currentPath + ".mxml")) {
329                     continue; // skip all applications if they are not needed
330                 }
331 
332                 String currentArtifactFileName = artefact.getFileName();
333 
334                 protocol.append("Starting " + artefact.getType() + " '" + artefact.getFileName() + "'...").append("\n");
335                 fireEvent(new CompilerEvent(this, "Starting...", CompilerEvent.EventType.Started, currentArtifactFileName));
336 
337                 File artifactFile = null;
338 
339                 if (artefact.getType() == Artefact.Type.css) {
340                     artifactFile = new File(projectDir, app.getName() + File.separator + Constants.ComponentFolders.CSS_DIR + File.separator + currentArtifactFileName);
341                 } else {
342                     artifactFile = new File(projectDir, app.getName() + File.separator + Constants.ComponentFolders.MXML_DIR + File.separator + currentArtifactFileName);
343                 }
344 
345                 if (!artifactFile.exists()) {
346                     return new CompilationResults("No changes: Artifact was deleted... Finishing.", false, false, project.getName(), component.getName());
347                 }
348 
349                 String applicationCacheKey = project.toString() + app.toString() + currentArtifactFileName;
350                 flex2.tools.oem.Application application = applicationCache.get(applicationCacheKey);
351 
352                 if (application == null) {
353                     try {
354                         application = new Application(artifactFile);
355                     } catch (FileNotFoundException e) {/* impossible! we already checked it */}
356 
357                     // FIXME: library cache
358 //                    if (libraryCache != null) {
359 //                        application.setSwcCache(libraryCache);
360 //                    }
361 
362                     if (conf.getBoolean(ConfigProperties.SHOW_PROGRESS, false)) {
363                         application.setProgressMeter(progressTracker);
364                     }
365 
366 
367                     // TODO: add another (simple) configuration for CSS artifacts
368                     flex2.tools.oem.Configuration configuration = application.getDefaultConfiguration();
369                     configure(configuration, new File(projectDir, app.getName()), new File(dstDir), app, componentDependencies.get(component), project, projectDir);
370                     application.setConfiguration(configuration);
371 
372                     application.setOutput(getArtifactOutputFile(artefact, new File(dstDir)));
373 
374                     applicationCache.put(applicationCacheKey, application);
375                 }
376 
377                 protocol.append("Compiling...").append("\n");
378                 fireEvent(new CompilerEvent(this, "Compiling...", CompilerEvent.EventType.Compilation, currentArtifactFileName));
379 
380                 if (outputFiles.size() == 0) {
381                     for (Artefact a : app.getArtefacts()) {
382                         outputFiles.add(getArtifactOutputFile(a, new File(dstDir)));
383                     }
384                 }
385 
386                 try {
387                     application.build(true);
388                 } catch (IOException e) {
389                     log.error("Error during copilation", e);
390                     protocol.append("Technical error: ").append(e.getMessage());
391                     return new CompilationResults(protocol.toString(), true, true, project.getName(), component.getName());
392                 }
393 
394                 // FIXME: library cache
395                 // libraryCache = application.getSwcCache();
396 
397                 report = application.getReport();
398                 currentBuildTarget = currentArtifactFileName;
399 
400                 if (hasErrors(report.getMessages())) {
401                     break;
402                 }
403             }
404         }
405 
406         long processingTime = System.currentTimeMillis() - startTime;
407 
408         if (hasErrors(report.getMessages())) {
409             if (outputFiles != null) {
410                 for (File f : outputFiles) {
411                     f.delete(); // delete old artefact
412                 }
413             }
414 
415             String s = "Flex compilation errors: \n" + toString(report.getMessages());
416             protocol.append(s).append("\n");
417             protocol.append("Compilation time: ").append(processingTime).append(" ms").append("\n");
418 
419             CompilerEvent event = new CompilerEvent(this, s, CompilerEvent.EventType.Errors, currentBuildTarget);
420             event.setCompilationTime(processingTime);
421             event.setCompilationProtocol(protocol.toString());
422             fireEvent(event);
423             return new CompilationResults(protocol.toString(), true, true, project.getName(), component.getName());
424         } else {
425             String s = "Flex compilation successful." + toString(report.getMessages());
426             protocol.append(s).append("\n");
427             protocol.append("Compilation time: ").append(processingTime).append(" ms").append("\n");
428 
429             CompilerEvent event = new CompilerEvent(this, s, CompilerEvent.EventType.Success, currentBuildTarget);
430             event.setCompilationTime(processingTime);
431             event.setCompilationProtocol(protocol.toString());
432             fireEvent(event);
433             return new CompilationResults(protocol.toString(), true, false, project.getName(), component.getName());
434         }

437     protected void addLibraryAssets(File baseDir, String currPath, Library library) {
438         for (File f : (currPath.equals("") ? baseDir : new File(baseDir, currPath)).listFiles()) {
439             if (f.isDirectory()) {
440                 addLibraryAssets(baseDir, currPath + File.separator + f.getName(), library);
441             } else {
442                 if (f.getName().endsWith(".css")) {
443                     library.addStyleSheet(f.getName(), f);
444                 } else if (f.getName().endsWith(".properties") && currPath.equals("")) {
445                     library.addResourceBundle(f.getName().substring(0, f.getName().lastIndexOf(".")));
446                 } else if (!f.getName().endsWith(".as")) {
447                     library.addArchiveFile(f.getName(), f);
448                 }
449             }
450         }
451     }

453     private boolean isModificationsPresent(File path, Date date) {
454         for (File f : path.listFiles()) {
455             boolean modified = false;
456 
457             if (f.isDirectory()) {
458                 modified = isModificationsPresent(f, date);
459             } else {
460                 if (!f.getName().endsWith(".cache")) {
461                     modified = new Date(f.lastModified()).after(date);
462                 }
463             }
464 
465             if (modified) {
466                 return true;
467             }
468 
469         }
470 
471         return false;
472     }

474     protected void configure(flex2.tools.oem.Configuration configuration, File componentBaseDir, File dstDir, Component component, Set<Component> componentDependencies, Project project, File projectDir) {
475         configuration.setConfiguration(new File(frameworksDir, "flex-config.xml"));
476         configuration.setLocalFontSnapshot(new File(frameworksDir, conf.getString(ConfigProperties.FLEX_FONTS, Constants.DEFAULT_FLEX_FONTS)));
477 
478         // Locale
479 
480         if (component.getLocales().size() > 0) {
481             configuration.setLocale(component.getLocales().toArray(new String[component.getLocales().size()]));
482         } else if (project.getDefaultLocales().size() > 0) {
483             configuration.setLocale(project.getDefaultLocales().toArray(new String[project.getDefaultLocales().size()]));
484         } else {
485             configuration.setLocale(Constants.DEFAULT_LOCALE);
486         }
487 
488         // Sources
489 
490         List<File> sourcePaths = new ArrayList<File>();
491         File mxmlPath = new File(componentBaseDir, Constants.ComponentFolders.MXML_DIR);
492         if (mxmlPath.exists()) {
493             sourcePaths.add(mxmlPath);
494         }
495         File srcPath = new File(componentBaseDir, Constants.ComponentFolders.SRC_DIR);
496         if (srcPath.exists()) {
497             sourcePaths.add(srcPath);
498         }
499         configuration.setSourcePath(sourcePaths.toArray(new File[sourcePaths.size()]));
500 
501         // Libraries
502         List<File> libs = new ArrayList<File>();
503 
504         libs.add(new File(frameworksDir, "libs"));
505         libs.add(new File(frameworksDir, "locale"));
506 
507         File libDir = new File(componentBaseDir, Constants.ComponentFolders.LIB_DIR);
508         if (libDir.exists()) {
509             libs.add(libDir);
510         }
511 
512         if (componentDependencies != null && componentDependencies.size() > 0) {
513             libs.add(new File(dstDir, Constants.ComponentFolders.SWC_TARGET_DIR));
514             for (Component c : componentDependencies) {
515                 File compLibDir = new File(projectDir, c.getName() + File.separator + Constants.ComponentFolders.LIB_DIR);
516                 if (compLibDir.exists()) {
517                     libs.add(compLibDir);
518                 }
519             }
520         }
521 
522         if (conf.isPresent(ConfigProperties.FLEX_ADDITIONAL_LIBS)) {
523             String[] additional = conf.getStringArray(ConfigProperties.FLEX_ADDITIONAL_LIBS);
524             for (String al : additional) {
525                 libs.add(new File(al));
526             }
527         }
528 
529         configuration.addLibraryPath(libs.toArray(new File[libs.size()]));
530 
531         // Themes
532 
533         if (component instanceof org.oleg.fcs.project.model.Application) {
534             List<Theme> themesDefifitions = ((org.oleg.fcs.project.model.Application) component).getThemes();
535             if (themesDefifitions == null || themesDefifitions.size() == 0) {
536                 themesDefifitions = project.getDefaultThemes();
537             }
538 
539             if (themesDefifitions != null && themesDefifitions.size() > 0) {
540                 List<File> themeFiles = new ArrayList<File>();
541 
542                 for (Theme theme : themesDefifitions) {
543                     switch (theme.getType()) {
544                         case component:
545                             themeFiles.add(new File(dstDir, Constants.ComponentFolders.SWC_TARGET_DIR + File.separator + theme.getName() + (theme.getName().endsWith(".swc") ? "" : ".swc")));
546                             break;
547                         case lib:
548                             String fileName = theme.getName() + (theme.getName().endsWith(".swc") ? "" : ".swc");
549                             boolean found = false;
550 
551                             File compLibDir = new File(componentBaseDir, Constants.ComponentFolders.LIB_DIR);
552                             if (compLibDir.exists() && hasFile(compLibDir, fileName)) {
553                                 themeFiles.add(new File(compLibDir, fileName));
554                                 found = true;
555                             } else {
556                                 for (Component c : componentDependencies) {
557                                     compLibDir = new File(projectDir, c.getName() + File.separator + Constants.ComponentFolders.LIB_DIR);
558                                     if (compLibDir.exists() && hasFile(compLibDir, fileName)) {
559                                         themeFiles.add(new File(compLibDir, fileName));
560                                         found = true;
561                                         break;
562                                     }
563                                 }
564                             }
565 
566                             if (!found) {
567                                 throw new ProjectException("Can't find theme library in the buid path: " + fileName);
568                             }
569                             break;
570                         case file:
571                             if (new File(theme.getName()).isAbsolute()) {
572                                 themeFiles.add(new File(theme.getName()));
573                             } else {
574                                 themeFiles.add(new File(componentBaseDir, theme.getName()));
575                             }
576                             break;
577                         case builtin:
578                             File themeDir = new File(frameworksDir, "themes" + File.separator + theme.getName());
579                             if (!themeDir.exists()) {
580                                 throw new ProjectException("Unable to find buitin theme with name " + theme.getName());
581                             }
582 
583                             boolean themeFileFound = false;
584                             for (File themeFile : themeDir.listFiles()) {
585                                 if (themeFile.isFile() && (themeFile.getName().endsWith(".css") || themeFile.getName().endsWith(".swc"))) {
586                                     themeFiles.add(themeFile);
587                                     themeFileFound = true;
588                                     break;
589                                 }
590                             }
591 
592                             if (!themeFileFound) {
593                                 throw new ProjectException("Unable to find any sutable theme files for the builtin theme " + theme.getName());
594                             }
595                             break;
596                     }
597                 }
598 
599                 configuration.addTheme(themeFiles.toArray(new File[themeFiles.size()]));
600             }
601         }
602 
603         // project and component flex config
604 
605         addFlexConfig(projectDir, project.getDefaultFlexConfig(), configuration);
606         addFlexConfig(componentBaseDir, component.getFlexConfig(), configuration);
607     }

621     protected void addFlexConfig(File basePath, String userPath, flex2.tools.oem.Configuration configuration) {
622         if (userPath != null && !userPath.trim().equals("")) {
623             if (new File(userPath).isAbsolute()) {
624                 configuration.addConfiguration(new File(userPath));
625             } else {
626                 configuration.addConfiguration(new File(basePath, userPath));
627             }
628         } else {
629             File defaultConfig = new File(basePath, Constants.DEFAULT_FLEX_CONFIG_FILE_NAME);
630             if (defaultConfig.exists()) {
631                 configuration.addConfiguration(defaultConfig);

660                 count++;

664                 } else if (count == line) {
665                     sb.append(String.format("%4d> %s\n", count, s));
